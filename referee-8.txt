\begin{lemma}[Cubic far–field bound with explicit constant]\label{lem:cubic-far}
Let \(I=[t_0-L,t_0+L]\), \(Q(\alpha I)=\{\,\tfrac12+\sigma+it:\ t\in I,\ 0\le\sigma\le \alpha L\,\}\) with \(\alpha\in[1,2]\).
For any \(\rho=\beta+i\gamma\) with \(r:=|t_0-\gamma|\ge L\) one has
\[
\frac{1}{|I|}\iint_{Q(\alpha I)} \frac{\sigma}{|(\tfrac12+\sigma+it)-\rho|^2}\,dt\,d\sigma
\ \le\ \frac{C_\alpha}{(r/L)^3},\qquad C_\alpha\le 0.0450.
\]
\emph{Proof.} It suffices to take the worst case \(\beta=\tfrac12\). Set \(a:=r-L\ge 0\), \(u:=|t-\gamma|\). Then for \(t\in I\),
\(u\ge a\) and
\[
\frac{1}{|I|}\!\iint\frac{\sigma}{u^2+\sigma^2}\,dt\,d\sigma
\ \le\ \frac{1}{2L}\!\int_0^{\alpha L}\!\!\Big( \int_{-L}^{L}\frac{\sigma}{(a+|x|)^2+\sigma^2}\,dx \Big)\,d\sigma
\ \le\ \frac{1}{L}\!\int_0^{\alpha L}\!\arctan\!\Big(\frac{\sigma}{a}\Big)\,d\sigma.
\]
Evaluating the antiderivative gives
\[
\frac{1}{L}\Big[\ \sigma\arctan\!\Big(\frac{\sigma}{a}\Big)\ -\ \frac{a}{2}\log\!\Big(1+\frac{\sigma^2}{a^2}\Big)\ \Big]_{0}^{\alpha L}
\ =:\ F\!\Big(\frac{\alpha L}{a}\Big).
\]
From the inequality \(\log(1+x)\ge \tfrac{2x}{2+x}\) and \(\arctan x\le x\), one checks that \(F(y)\le \tfrac{9}{200}\,y^3\) for all \(y\in[0,2]\).
Since \(a=r-L\) and \(y=\frac{\alpha L}{a}\le \frac{2}{j-1}\) on the \(j\)-th annulus (\(r\in[jL,(j+1)L)\)), the claim follows with \(C_\alpha=\tfrac{9}{200}\le 0.0450\).
\qedhere
\end{lemma}

\begin{lemma}[Removable singularity for bounded Schur quotients]\label{lem:removable}
Let $D\subset\C$ be a disk and $F:D\setminus\{\rho\}\to\C$ be analytic with $\bigl|\frac{F-1}{F+1}\bigr|\le 1$ on $D\setminus\{\rho\}$.
Then $F$ extends holomorphically to $D$.
\end{lemma}
\begin{proof}
The Cayley transform $G=(F-1)/(F+1)$ is bounded by $1$ on the punctured disk, hence by Riemann’s theorem the singularity at $\rho$ is removable for $G$. The inverse Cayley map is analytic on $\{|G|<1\}$, so $F$ extends holomorphically to $D$.
\end{proof}


Then cite Lemma \ref{lem:removable} in the proof of Theorem \ref{thm:globalize-RH}.
