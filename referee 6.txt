\subsection*{Product certificate yields (P+)}


\begin{theorem}[Length–free product certificate]\label{thm:prod-cert}
Fix a mass–1 window $\psi$ with $\psi\equiv 1$ on $[-1,1]$ and $\mathrm{supp}\,\psi\subset[-2,2]$, and set
$\varphi_{L,t_0}(t)=\psi\big((t-t_0)/L\big)/L$ so that $\int \varphi_{L,t_0}=1$ and $\mathrm{supp}\,\varphi_{L,t_0}\subset I$.
Then for every interval $I$ and every $L>0$, $t_0\in I$,
\begin{equation}\label{eq:prod-certificate}
  c_0(\psi)\,\mu\!\big(Q(I)\big)
  \;\le\; \int_{\mathbb R} \!\big(-w'(t)\big)\,\varphi_{L,t_0}(t)\,dt
  \;\le\; C_H(\psi)\,M_\psi \;+\; C_P(\kappa).
\end{equation}
All constants depend only on the fixed template $\psi$ (and $\kappa$ for the bandlimit term), not on $I,L,t_0$.
\end{theorem}


\begin{proof}[Proof sketch]
Pair the phase–velocity identity with $\varphi_{L,t_0}$. On $Q(I)$ the Poisson kernel against a mass–1 plateau window
has a uniform pointwise lower bound $c_0(\psi)$, giving the left inequality. For the right inequality:
integration by parts yields
$\big|\langle \mathcal H[\varphi_{L,t_0}]',\,u\rangle\big|\le \|\mathcal H[\varphi_{L,t_0}]'\|_\infty \int_I |u-\ell_I|
\le \frac{C_H(\psi)}{L}\,\big(M_\psi\,L\big)=C_H(\psi)M_\psi$,
while the bandlimited tail is $\le C_P(\kappa)$ by the Dirichlet–Parseval bound.
Our support choice ensures no off–box leakage, so no additional tail terms appear.
\end{proof}


\begin{corollary}[(P+) from the product certificate]\label{cor:wedge}
Let $\Theta:=(C_H(\psi)M_\psi+C_P(\kappa))/c_0(\psi)$. Then from \eqref{eq:prod-certificate},
\[
  \mu\!\big(Q(I)\big)\ \le\ \Theta
  \quad\Rightarrow\quad
  \int_I (-w') \ \le\ \pi\,\Theta.
\]
If $\Theta\le\tfrac12$, then $\int_I (-w')\le \tfrac{\pi}{2}$ for every interval $I$, hence the boundary wedge \emph{(P+)}
holds; consequently $2\mathcal J$ is Herglotz on $\Omega$ and $\Theta=(2\mathcal J-1)/(2\mathcal J+1)$ is Schur.
\end{corollary}


\noindent\textit{Route declaration.} We prove \emph{(P+)} only by Theorem~\ref{thm:prod-cert} and Corollary~\ref{cor:wedge}.
The PSC/Carleson density discussion is archived and not used in the main chain.
