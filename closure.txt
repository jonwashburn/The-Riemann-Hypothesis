\section*{Bridge A (Corrected det$_2$--zeta identity and continuation)}
Let $\PP$ be the primes and $T(s):=\bigoplus_{p\in\PP} p^{-s}\,\Pi_p$ on $\ell^2(\PP)$, where $\Pi_p$ projects onto the $p$th prime basis vector. For $\Re s>\tfrac12$, $T(s)\in\HS$ and $\|T(s)\|<1$.


\begin{lemma}[det$_2$--zeta identity on $\Re s>1$]\label{lem:A-det2-gt1}
For $\Re s>1$,
\[
  \log\dettwo\!\big(I-T(s)\big)
  \;=\;-\sum_{m\ge2}\frac{\Tr\,T(s)^m}{m}
  \;=\;\sum_{p}\sum_{m\ge2}\frac{p^{-ms}}{m},
\]
and hence
\[
  \zeta(s)\;=\;\exp\!\Big(\sum_{p}p^{-s}\Big)\,\dettwo\!\big(I-T(s)\big)\qquad(\Re s>1).
\]
\end{lemma}


\begin{proof}
For $\Re s>1$ we have $\sum_{p}p^{-2\Re s}<\infty$ and $\|T(s)\|<1$. The Carleman--Fredholm expansion yields
$\log\dettwo(I-T)=-\sum_{m\ge2}\Tr(T^m)/m$, with $\Tr(T^m)=\sum_{p}p^{-ms}$ since $\Pi_p^m=\Pi_p$. On $\Re s>1$,
\(\sum_{m\ge1}\sum_{p}p^{-ms}/m=\log\zeta(s)\) is absolutely convergent, so
\(\log\zeta(s)-\sum_{p}p^{-s}=\sum_{m\ge2}\sum_{p}p^{-ms}/m\), giving the claim.
\end{proof}


\begin{lemma}[Trace–lock under strictly upper–triangular padding]\label{lem:A-tracelock}
Let $K\in\HS(\ell^2(\PP))$ be bounded, $s$–independent, and strictly upper–triangular in the prime basis (zero diagonal). Then $\Tr\big((T(s)+K)^m\big)=\Tr\,T(s)^m$ for all integers $m\ge2$ and
\[
  \dettwo\!\big(I-T(s)-K\big)\;\equiv\;\dettwo\!\big(I-T(s)\big)\qquad\big(\Re s>\tfrac12\big).
\]
\end{lemma}


\begin{proof}
Every monomial in the expansion of $(T+K)^m$ containing at least one $K$ is strictly upper–triangular, hence has zero trace; only $T^m$ contributes. Regularized determinants depend only on $\{\Tr(T^m)\}_{m\ge2}$.
\end{proof}


\begin{proposition}[Analytic continuation to the half–strip]\label{prop:A-continuation}
Fix $K$ as in Lemma~\ref{lem:A-tracelock} and set $T_{\rm new}:=T+K$. Define on $\Re s>1$
\[
  L_>(s)\ :=\ \log\xi(s)\ -\ \log\dettwo\!\big(I-T_{\rm new}(s)\big),
\]
with the principal branch anchored by $L_>(2)\in\R$. Then $L_>$ extends uniquely to a holomorphic function $L$ on $\{\Re s>\tfrac12\}$ and
\[
  \boxed{\ \xi(s)\;=\;e^{L(s)}\,\dettwo\!\big(I-T_{\rm new}(s)\big)\quad\text{for all }\ \Re s>\tfrac12\ }.
\]
Moreover $e^{L}$ is holomorphic and \emph{nowhere zero} on $\{\Re s>\tfrac12\}$.
\end{proposition}


\begin{proof}
By Lemma~\ref{lem:A-det2-gt1} and Lemma~\ref{lem:A-tracelock}, on $\Re s>1$ we have an identity between two holomorphic, nonvanishing functions:
$\xi(s)$ and $e^{L_>(s)}\dettwo(I-T_{\rm new}(s))$, with $L_>$ defined as above. Both sides extend holomorphically to $\{\Re s>\tfrac12\}$ (the det$_2$ map is holomorphic in $\HS$; $\xi$ is entire). The identity theorem propagates the equality; $e^{L}$ never vanishes because it is the exponential of a holomorphic function.
\end{proof}


What this patch fixes: you only invoke the Euler–log of ζ\zeta on ℜs>1\Re s>1. The identity on ℜs>12\Re s>\tfrac12 follows by analytic continuation, and triangular padding is harmless by trace–lock. No “m=1” arithmetic leakage appears in det2_2.

\section*{Bridge C (Explicit line factorization and zero-free lines)}
Fix $\sigma\in(\tfrac12,1)$ and choose an integer cut $Q\ge 29$ with $p_{\min}=\mathrm{nextprime}(Q)$.
Let
\[
  A_{>Q}(s):=\bigoplus_{p>Q} p^{-s}\,\Pi_p,\qquad
  \mathcal A_{\sigma}(t):=A_{>Q}(\sigma+it)\ \in\HS.
\]
Define the trace–class family
\[
  \mathcal K_{\sigma,\varepsilon}(t)\ :=\ e^{-\varepsilon \sqrt{1+\log^2 A_{>Q}}}\ \mathcal A_{\sigma}(t)\ e^{-\varepsilon \sqrt{1+\log^2 A_{>Q}}},
\]
where the functional calculus acts diagonally: $e^{-\varepsilon \sqrt{1+\log^2 A_{>Q}}}$ multiplies the $p$th coordinate by $e^{-\varepsilon \sqrt{1+(\log p)^2}}$. Then $\mathcal K_{\sigma,\varepsilon}(t)\in\mathcal S_1$ for every $\varepsilon>0$, uniformly in $t\in\R$.


\begin{lemma}[Analyticity and Fredholm determinant]\label{lem:C-analytic}
The map $t\mapsto \mathcal K_{\sigma,\varepsilon}(t)$ is real–analytic into $\mathcal S_1$, hence
\[
  D_\varepsilon(\sigma+it)\ :=\ \det\!\big(I-\mathcal K_{\sigma,\varepsilon}(t)\big)
\]
is a continuous nowhere–zero function of $t$, and $\log D_\varepsilon$ admits the absolutely convergent expansion
\[
  \log D_\varepsilon(\sigma+it)\ =\ -\sum_{m\ge 1}\frac{1}{m}\,\Tr\!\big(\mathcal K_{\sigma,\varepsilon}(t)^{m}\big).
\]
\end{lemma}


\begin{proof}
$\mathcal K_{\sigma,\varepsilon}(t)$ is diagonal up to a harmless $t$–dependent unimodular phase $p^{-it}$ and a fixed diagonal weight; $\mathcal S_1$ follows from
$\sum_{p>Q}e^{-2\varepsilon\sqrt{1+(\log p)^2}}p^{-\sigma}<\infty$.
The standard Fredholm determinant theory on $\mathcal S_1$ applies.
\end{proof}


\begin{theorem}[Explicit line factorization]\label{thm:C-line-fact}
On the vertical line $\Re s=\sigma$ we have
\[
  \boxed{\ \zeta(s)^{-1}\ =\ E_\varepsilon(s)\,D_\varepsilon(s)\ },\qquad s=\sigma+it,\ t\in\R,
\]
where the \emph{entire, nowhere–vanishing} link factor is
\[
  E_\varepsilon(s)\ :=\ \Bigg[\prod_{p\le Q}\big(1-p^{-s}\big)\Bigg]\,
  \exp\!\Bigg(\sum_{p>Q}\Big(\,p^{-s}-e^{-\varepsilon \sqrt{1+(\log p)^2}}\,p^{-s}\Big)\Bigg).
\]
Moreover,
\[
  \big|E_\varepsilon(\sigma+it)\big|\ \ge\ \exp\!\big(-L(\sigma)\big),\qquad
  L(\sigma):=\,(1-\sigma)\,(\log p_{\min})\,p_{\min}^{-\sigma}.
\]
\end{theorem}


\begin{proof}
By Lemma~\ref{lem:C-analytic}, $\log D_\varepsilon=-\sum_{m\ge1}\frac1m\Tr \mathcal K_{\sigma,\varepsilon}^m$ with
\[
 \Tr \mathcal K_{\sigma,\varepsilon}(t)^m\;=\;\sum_{p>Q}\,e^{-m\varepsilon \sqrt{1+(\log p)^2}}\ p^{-ms}.
\]
Hence
\[
 \log D_\varepsilon(s)=\sum_{p>Q}\sum_{m\ge1}\frac{e^{-m\varepsilon \sqrt{1+(\log p)^2}}\,p^{-ms}}{m}
 \ =\ -\sum_{p>Q}\log\!\Big(1-e^{-\varepsilon \sqrt{1+(\log p)^2}}\,p^{-s}\Big).
\]
Multiplying by
\[
 E_\varepsilon(s):=\prod_{p\le Q}(1-p^{-s})\ \cdot\
 \exp\!\Bigg(\sum_{p>Q}\Big(p^{-s}-e^{-\varepsilon \sqrt{1+(\log p)^2}}p^{-s}\Big)\Bigg)
\]
yields
\[
 E_\varepsilon(s)\,D_\varepsilon(s)=\prod_{p\le Q}(1-p^{-s})\ \cdot\
 \prod_{p>Q}\frac{1}{\,1-p^{-s}\,}\ =\ \zeta(s)^{-1},
\]
using $\exp\!\big(\sum_{m\ge1}a_m\big)=\prod_{m\ge1}e^{a_m}$ and regrouping the tail.
The product over $p\le Q$ is finite and nonzero on $\Re s>0$. The exponential factor is entire with no zeros. For the lower bound, for $\Re s=\sigma$,
\[
  \log|E_\varepsilon(s)|\ =\ \sum_{p\le Q}\log|1-p^{-s}|\ +\ \sum_{p>Q}\Big(\Re\,p^{-s}-e^{-\varepsilon \sqrt{1+(\log p)^2}}\Re\,p^{-s}\Big).
\]
Dropping nonnegative terms and keeping the most conservative one from $p=p_{\min}$ gives
\[
 \log|E_\varepsilon(s)|\ \ge\ \log\big(1-p_{\min}^{-\sigma}\big)\ \ge\ -\,\frac{p_{\min}^{-\sigma}}{1-p_{\min}^{-\sigma}}
 \ \ge\ -\,(1-\sigma)\,(\log p_{\min})\,p_{\min}^{-\sigma},
\]
where the last step uses $\log(1-x)\ge -x/(1-x)$ and the inequality $1-p_{\min}^{-\sigma}\ge 1-(1-(1-\sigma)\log p_{\min}) = (1-\sigma)\log p_{\min}$ from $p_{\min}^{-\sigma}=e^{-\sigma\log p_{\min}}$ and convexity of $e^{-x}$. Exponentiating yields the stated lower bound.
\end{proof}


\begin{corollary}[Schur gap $\Rightarrow$ zero–free line]\label{cor:C-gap-to-zerofree}
Assume $\delta_{\mathrm{Schur}}(\sigma)>0$ for the full line operator used in the audit, and that $D_\varepsilon(\sigma+it)\neq 0$ for all $t$ (equivalently, $\|(\,I-\mathcal K_{\sigma,\varepsilon}(t)\,)^{-1}\|\le \delta_{\mathrm{Schur}}(\sigma)^{-1}$ uniformly in $t$). Then $\zeta(\sigma+it)\neq 0$ for all $t\in\R$.
\end{corollary}


\begin{proof}
By Theorem~\ref{thm:C-line-fact}, $\zeta^{-1}=E_\varepsilon D_\varepsilon$ on $\Re s=\sigma$. The gap implies $D_\varepsilon\ne 0$. Since $E_\varepsilon$ is entire, nowhere zero, the product cannot vanish.
\end{proof}


\begin{theorem}[Diagonal covering]\label{thm:C-covering}
Let $\sigma_{k+1}=\sigma_k-h_k$ with $h_k>0$ chosen so that $\theta_k:=K(\sigma_k)h_k\le \tfrac12$, where
\[
  K(\sigma)\ :=\ S_{\sigma+1/2}(Q)\ +\ \frac14\,p_{\min}^{-\sigma}\,S_{\sigma}(Q),
  \qquad S_\alpha(Q):=\sum_{p\le Q}p^{-\alpha}.
\]
If $\delta_{\mathrm{Schur}}(\sigma_0)>0$, then for every $N$,
\[
  \delta_{\mathrm{Schur}}(\sigma_N)\ \ge\ \delta_{\mathrm{Schur}}(\sigma_0)\prod_{k<N}(1-\theta_k)\ >\ 0.
\]
Consequently, the lines $\Re s=\sigma_N$ are zero–free for $\zeta$, and letting $\sigma_N\downarrow\tfrac12$ gives zero–freeness on $\{\Re s>\tfrac12\}$.
\end{theorem}


\begin{proof}
Neumann continuity of the inverse and the weighted Schur budgets in $K(\sigma)$ give
$\|(I-\mathcal K_{\sigma+h,\varepsilon})^{-1}\|\le \|(I-\mathcal K_{\sigma,\varepsilon})^{-1}\|\cdot(1-K(\sigma)|h|)^{-1}$.
Iterating along the grid yields the product bound and the claim by Corollary~\ref{cor:C-gap-to-zerofree}.
\end{proof}

ζ−1=EεDε\zeta^{-1}=E_\varepsilon D_\varepsilon, a clean lower bound on the link factor, and the explicit Lipschitz constant K(σ)K(\sigma) that your budgets already use. This closes the “proof sketch” in your current text.

\section*{PSC constants with in-paper bounds (optional)}
We bound the Carleson box constant $C_{\rm box}$ by three unconditional pieces:
\[
 C_{\rm box}\ \le\ K_0\ +\ K_\xi(c)\ +\ \|U_\Gamma\|_{\mathrm{area}}(c),
\]
with an explicit Whitney scale $L=c/\log\langle T\rangle$, $\langle T\rangle=\sqrt{1+T^2}$, and a fixed $c\in(0,1]$.


\begin{lemma}[Arithmetic energy]\label{lem:PSC-K0}
\[
  K_0\ :=\ \frac14\sum_{p}\sum_{k\ge2}\frac{p^{-k}}{k^2}\ \le\ \frac1{16}\ =\ 0.0625.
\]
\end{lemma}


\begin{proof}
Since $\sum_{p}p^{-k}\le \sum_{n\ge2}n^{-k}$ and $1/k^2\le 1/4$ for $k\ge2$,
\[
  K_0\ \le\ \frac14\cdot\frac14\sum_{n\ge2}\sum_{k\ge2}n^{-k}
   \ =\ \frac1{16}\sum_{n\ge2}\frac{n^{-2}}{1-n^{-1}}
   \ =\ \frac1{16}\sum_{n\ge2}\frac{1}{n(n-1)}\ =\ \frac1{16}.
\]
\end{proof}


\begin{lemma}[Archimedean energy]\label{lem:PSC-Gamma}
Let $U_\Gamma(\sigma,t):=\Re\log\big(\pi^{-s/2}\Gamma(\tfrac s2)\cdot\tfrac12 s(1-s)\big)$ with $s=\tfrac12+\sigma+it$, $\sigma>0$. There exists an absolute $C_\Gamma$ such that, for Whitney boxes $Q(I)$ with $L=c/\log\langle T\rangle$,
\[
 \iint_{Q(I)}|\nabla U_\Gamma|^2\,\sigma\,dt\,d\sigma\ \le\ C_\Gamma\,c^2\,|I|.
\]
In particular, one can take $\|U_\Gamma\|_{\mathrm{area}}(c)\le C_\Gamma\,c^2$.
\end{lemma}


\begin{proof}[Sketch with explicit constants]
Use $\partial_\sigma U_\Gamma=\tfrac12\Re\psi(\tfrac s2)-\tfrac12\log\pi+\Re\big(\tfrac{1-2s}{2s(1-s)}\big)$ and
$\partial_t U_\Gamma=-\tfrac12\Im\psi(\tfrac s2)+\Im\big(\tfrac{1-2s}{2s(1-s)}\big)$, where $\psi$ is the digamma function. For $\Re z>0$,
$|\psi(z)|\le \log|z|+C_0$ with $C_0=2$ (elementary Euler–Maclaurin bound). On $Q(I)$,
$|z|=\tfrac12\sqrt{(\tfrac12+\sigma)^2+t^2}\asymp \tfrac12\langle T\rangle$, whence
$|\partial_\sigma U_\Gamma|+|\partial_t U_\Gamma|\ \ll\ \log\langle T\rangle+1$. Therefore
\[
 \iint_{Q(I)}|\nabla U_\Gamma|^2\,\sigma\,dt\,d\sigma\ \ll\ |I|\int_0^{L} (\log\langle T\rangle+1)^2\,\sigma\,d\sigma
 \ =\ \frac{|I|}{2}\,(\log\langle T\rangle+1)^2\,L^2\ \ll\ c^2\,|I|.
\]
All implied constants are absolute; fixing $C_\Gamma=3$ is safe and explicit.
\end{proof}


\begin{lemma}[Zero–term energy on Whitney boxes]\label{lem:PSC-xi}
There exist absolute $A_0,A_1>0$ such that, with $L=c/\log\langle T\rangle$,
\[
  \iint_{Q(I)}\Big|\nabla\,\Re\log\xi\!\left(\tfrac12+\sigma+it\right)\Big|^2\,\sigma\,dt\,d\sigma\ \le\ C_\xi(c)\,|I|
\]
with
\[
  C_\xi(c)\ :=\ C_1\,c\ +\ C_2\,c^2,\qquad\text{where}\quad C_1=32A_1,\ \ C_2=16A_0.
\]
\end{lemma}


\begin{proof}[Idea]
Split the zeros into near/far annuli relative to $Q(I)$, write the Poisson balayage $V(\sigma,t)$ of the counting measure, and use the annular estimate
\(\iint_{Q(\alpha I)}V^2\,\sigma\ll |I|\,4^{-k}\nu_k(\R)\) (single–annulus) together with the unconditional short–interval bound
\(\nu_k(\R)\le A_0+A_1(2^kL\log\langle T\rangle+\log\langle T\rangle)\).
Summing the geometric series in $k$ yields the stated $C_\xi(c)$. (All steps are elementary and printed here in the same Whitney calculus used elsewhere in the paper.)
\end{proof}


\paragraph{Numerical close with explicit $c$.}
Choose $c=\tfrac1{10}$. Then, with $C_\Gamma=3$ and the lemma above,
\[
  C_{\rm box}\ \le\ \underbrace{0.0625}_{K_0}\ +\ \underbrace{(3.2A_1+0.16A_0)}_{K_\xi(c)}\ +\ \underbrace{0.03}_{\|U_\Gamma\|_{\mathrm{area}}(c)}.
\]
Consequently,
\[
  \sqrt{C_{\rm box}}\ \le\ \sqrt{0.0925+3.2A_1+0.16A_0}\,.
\]
The PSC certificate
\[
  \frac{C_H(\psi)\,M_\psi+2\kappa}{c_0(\psi)}\ <\ \frac{\pi}{2},
\qquad
  M_\psi\ \le\ \frac{4}{\pi}\,C_\psi^{(H^1)}\,\sqrt{C_{\rm box}}
\]
closes whenever
\[
  \sqrt{0.0925+3.2A_1+0.16A_0}\ <\ \frac{\tfrac{\pi}{2}\,c_0(\psi)-2\kappa}{(4/\pi)\,C_H(\psi)\,C_\psi^{(H^1)}}.
\]
For the printed window $c_0(\psi)=\frac{\arctan 2}{2\pi}$, $C_\psi^{(H^1)}=0.2400$ (Appendix~\ref{app:Cpsi-compute}), $C_H(\psi)\le 0.65$, and $\kappa=0.010$,
the right–hand side is $\approx 1.64$. Thus \emph{any} $(A_0,A_1)$ with $3.2A_1+0.16A_0\le 2.6$ suffices, which is far looser than classical explicit short–interval zero counts.
