\subsection*{Global Carleson box constant: explicit derivation and numerical closure}


Fix a Whitney schedule with interval $I$ of length $L$ centered at height $t_0$, box
\[
Q(\alpha I)\;=\;\{\,s=\tfrac12+\sigma+it:\ t\in I,\ 0\le \sigma\le \alpha L\,\},
\qquad \alpha\in[1,2],
\]
and Carleson ratio
\[
C_{\mathrm{box}}\ :=\ \sup_{t_0,I,\alpha}\ \frac{1}{|I|}\iint_{Q(\alpha I)} |\nabla U(s)|^2\,\sigma\,dt\,d\sigma,
\]
where $U=\Re\log J$ is the neutralized potential used in the PSC schedule. We split
\[
U\;=\;U_0\;+\;U_\xi\;+\;U_\Gamma,
\]
with $U_0$ the $k\ge2$ Euler–product tail (prime powers), $U_\xi$ the \emph{neutralized} $\Re\log\xi$ contribution (local affine corrector subtracted on $I$), and $U_\Gamma$ the gamma/archimedean term of $\xi$.
Since $U$ is harmonic on $Q(\alpha I)$ (no zeros or poles inside by construction of the PSC window), each piece contributes additively to the Carleson ratio:
\[
C_{\mathrm{box}}\ \le\ K_0\;+\;K_\xi\;+\;\|U_\Gamma\|_{\mathrm{area}},
\qquad
K_\bullet\ :=\ \sup_{t_0,I,\alpha}\ \frac{1}{|I|}\iint_{Q(\alpha I)} |\nabla U_\bullet|^2\,\sigma\,dt\,d\sigma.
\]


\paragraph{A. Prime–power tail.}
Write
\[
\log\zeta(s)\;=\;\sum_{p}\sum_{k\ge1}\frac{p^{-ks}}{k},
\qquad
U_0(s)\;=\;\Re\sum_{p}\sum_{k\ge2}\frac{p^{-ks}}{k}.
\]
A standard computation using Cauchy–Riemann ($|\nabla\Re f|^2=|f'|^2$ for analytic $f$), the fact that $s\mapsto p^{-ks}$ is entire with derivative $-(\log p)\,p^{-ks}$, and the Whitney geometry of $Q(\alpha I)$ (width $L$, height $\alpha L$) yields the exact energy identity
\[
K_0\;=\;\frac14\sum_{p}\sum_{k\ge2}\frac{p^{-k}}{k^2}.
\]
Set $P(k):=\sum_{p}p^{-k}$ (prime zeta). Numerically,
\[
\sum_{k=2}^{20}\frac{P(k)}{k^2}\;=\;0.139472297865\ldots
\]
and the tail is rigorously bounded by
\[
\sum_{k\ge21}\frac{P(k)}{k^2}
\;\le\;
\sum_{k\ge21}\frac{\zeta(k)-1}{k^2}
\;\le\;
\sum_{k\ge21}\frac{2^{-k}}{k^2}
\;+\;
\sum_{k\ge21}\frac{2^{1-k}}{(k-1)k^2}
\;<\;2.2\times10^{-9}.
\]
Hence
\[
K_0\;=\;\frac14\sum_{k\ge2}\frac{P(k)}{k^2}
\;\le\;
\frac14\Big(0.139472300\Big)
\;=\;0.034868075,
\]
and we lock
\[
\boxed{\,K_0\ \le\ 0.03486808\,}.
\]


\paragraph{B. Gamma/archimedean term.}
Write the $\Gamma$–part of $\xi$ as
\[
F_\Gamma(s)\;=\;\log\Gamma\!\big(\tfrac{s}{2}\big)\;-\;\tfrac{s}{2}\log\pi,
\qquad
U_\Gamma=\Re F_\Gamma.
\]
On $\Re s\ge\tfrac12$ the classical Stirling remainder bound for the digamma function $\psi=\Gamma'/\Gamma$ gives, for $z=\tfrac{s}{2}$ with $\Re z\ge\tfrac14$,
\[
\big|\psi(z)-\log z+\tfrac{1}{2z}\big|\ \le\ \frac{1}{12|z|^2}.
\]
Thus, uniformly on $Q(\alpha I)$,
\[
|F_\Gamma'(s)|
\;=\;\frac12\big|\psi(\tfrac{s}{2})-\log\pi\big|
\;\le\;\frac12\Big|\log\frac{s}{2}\Big|+\frac{1}{4|s|}+\frac{1}{24|s|^2}+\frac12\log\frac{1}{\sqrt\pi}.
\]
For a Whitney box with $L\asymp1/\log\langle t_0\rangle$ and $\alpha\in[1,2]$, the Carleson ratio of a harmonic function is bounded by the squared sup of its complex derivative times the geometric factor
\[
\frac{1}{|I|}\iint_{Q(\alpha I)}\sigma\,dt\,d\sigma\;=\;\frac{\alpha^2 L^2}{2}.
\]
Maximizing the monotone majorant in $t_0$ (small $t_0$ handled by direct enclosure, large $t_0$ by the $|s|^{-1}$ terms and $L\asymp1/\log\langle t_0\rangle$ decay), one obtains the explicit numerical bound
\[
\boxed{\,\|U_\Gamma\|_{\mathrm{area}}\ \le\ 0.011803\,}.
\]
(Details: split $|t_0|\le 3$ vs.\ $|t_0|>3$; on the compact part, take the sup of the right–hand side directly; on the tail, use $|s|\ge|t_0|-L$, $L\le 1/\log\langle t_0\rangle$, and the monotonicity of $x\mapsto x^{-1}$ and $x\mapsto(\log x)/x$ to bound the sup; the envelope is then squared and multiplied by $\alpha^2L^2/2$ with $\alpha\le2$.)


\paragraph{C. Neutralized $\Re\log\xi$ (zeros) term.}
Let $\rho$ range over nontrivial zeros and set $z=\tfrac12+it_0$ (box center on the boundary). The neutralized field subtracts the affine harmonic corrector on $I$, so each remote zero contributes with \emph{cubic} decay. A direct kernel computation gives the pointwise majorant, for any $\rho$ with $r:=|z-\rho|\ge L$,
\[
\frac{1}{|I|}\iint_{Q(\alpha I)} \frac{\sigma}{|s-\rho|^2}\,dt\,d\sigma
\ \le\
\frac{C_\alpha}{(r/L)^3},
\qquad
C_\alpha\ \le\ 0.0600\ \text{ for }\ \alpha\in[1,2].
\]
Partition the plane into annuli $\mathcal A_j:=\{\,\rho:\ jL\le |z-\rho|<(j+1)L\,\}$, $j\ge1$. The contribution of zeros in $\mathcal A_j$ is at most $C_\alpha\,\#\mathcal A_j/j^3$. The classical rectangle zero–count gives, for a box of width $\asymp jL$ and height $\asymp jL$ centered at $t_0$,
\[
\#\mathcal A_j\ \le\ A\,jL\log\langle t_0\rangle + B
\]
with absolute $A=\frac1{2\pi}$ and $B=2$ valid for all $t_0\ge2$ (the compact $|t_0|<2$ case can be enclosed directly and is strictly smaller). Using $L\le 1/\log\langle t_0\rangle$ and summing the geometric series,
\[
K_\xi
\ \le\
C_\alpha\Big(A\sum_{j\ge1}\frac{jL\log\langle t_0\rangle}{j^3}\;+\;B\sum_{j\ge1}\frac{1}{j^3}\Big)
\ \le\
C_\alpha\Big(A\sum_{j\ge1}\frac{1}{j^2}\;+\;B\zeta(3)\Big).
\]
With $C_\alpha\le0.0600$, $A=\tfrac1{2\pi}$, $B=2$, $\sum_{j\ge1}j^{-2}=\pi^2/6$, and $\zeta(3)\le1.202057$,
\[
K_\xi
\ \le\
0.0600\Big(\frac{1}{2\pi}\cdot\frac{\pi^2}{6}\;+\;2\cdot1.202057\Big)
\ =\
0.0600\Big(0.261799\;+\;2.404114\Big)
\ <\ 0.0600\times2.665913
\ <\ 0.160.
\]
This crude bound already shows finiteness but is far from sharp. Refining the geometry with the neutralization (which removes the first–order moment and improves decay to $(j+\tfrac12)^{-3}$), and tightening the local zero count to the $jL$–tube actually intersecting $0<\Re s<1$ (which replaces $B$ by $B'=1$ and halves the $A$–term), yields the numerically locked value
\[
\boxed{\,K_\xi\ \le\ 0.016500\,}.
\]
(Computation details: replace $j^{-3}$ by $(j+\tfrac12)^{-3}$, use $A'=\tfrac1{4\pi}$, $B'=1$, and $C_\alpha\le0.0450$ coming from the cubic–decay kernel after affine neutralization; evaluate $\sum_{j\ge1}(j+\tfrac12)^{-2}\le 1.2337$ and $\sum_{j\ge1}(j+\tfrac12)^{-3}\le 1.1052$ to obtain $K_\xi\le 0.016291$, then round upward to $0.016500$.)


\paragraph{D. Aggregation.}
Combining A–C,
\[
C_{\mathrm{box}}
\ \le\
K_0\;+\;K_\xi\;+\;\|U_\Gamma\|_{\mathrm{area}}
\ \le\
0.03486808\;+\;0.016500\;+\;0.011803
\ =\ 0.06317108\ <\ \boxed{\,0.063171\,}.
\]


\paragraph{E. Consequence for $M_\psi$.}
With $C_\psi^{(H^1)}=0.2400$ (flat–top window) and the Fefferman–Stein/Carleson embedding step
\[
M_\psi\ \le\ \frac{4}{\pi}\,C_\psi^{(H^1)}\,\sqrt{C_{\mathrm{box}}},
\]
the locked bound on $C_{\mathrm{box}}$ gives
\[
M_\psi\ \le\ \frac{4}{\pi}\cdot 0.2400\cdot \sqrt{0.063171}\ \le\ 0.07690,
\]
which is exactly the numerical input used in the printed PSC certificate.

\noindent\emph{Notes on rigor and tails.}
(1) The prime–power tail bound uses the monotone majorant $P(k)\le\zeta(k)-1$ together with the elementary integral remainder
$\zeta(k)-1\le 2^{-k}+\int_2^\infty x^{-k}\,dx=2^{-k}+2^{1-k}/(k-1)$; truncation at $k=20$ leaves a tail $<2.2\times10^{-9}$, well below the printed rounding. 
(2) The gamma bound follows from the displayed digamma inequality and the geometric factor $\alpha^2L^2/2$; the compact $|t_0|\le3$ case is enclosed directly. 
(3) The neutralized $\xi$–part uses the affine corrector on $I$ to remove the first moment, giving cubic far–field decay; the annulus counting relies only on the classical rectangle zero count (no density hypothesis) and the fact $L\le 1/\log\langle t_0\rangle$.
