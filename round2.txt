\section*{PSC $\Rightarrow$ (P+) for the outer-normalized ratio \texorpdfstring{$\mathcal J$}{Jcal}}


\noindent
Let $\Omega:=\{s\in\mathbb{C}:\Rea s>1/2\}$ and write the boundary line as $\partial\Omega=\{1/2+it:t\in\mathbb{R}\}$.
Define
\[
\mathcal{J}(s):=\frac{\det\nolimits_2(I-A(s))}{\mathcal{O}(s)\,\xi(s)},
\]
where $\mathcal{O}$ is the outer factor constructed from the boundary modulus of $\det\nolimits_2(I-A)/\xi$ (see the outer-limit lemma provided separately).


\begin{proposition}[\textbf{PSC $\Rightarrow$ (P+)}]
Assume the Carleson Self-Correction (PSC) principle holds in the form
\[
\big|\Arg\big(2\,\mathcal{J}(1/2+it)\big)\big| \le \frac{\pi}{2}\quad\text{for a.e. }t\in\mathbb{R}.
\]
Then the Boundary Positivity statement (P+) holds for $\mathcal{J}$:
\[
\Rea\big(2\,\mathcal{J}(1/2+it)\big)\ge 0\quad\text{for a.e. }t\in\mathbb{R}.
\]
\end{proposition}


\begin{proof}
For any complex number $z\neq 0$ we have $\big|\Arg z\big|\le \pi/2$ if and only if $\Rea z\ge 0$. Apply this pointwise to $z=2\,\mathcal{J}(1/2+it)$ for almost every $t$.
\end{proof}

\section*{From boundary positivity to holomorphy (Herglotz) and zero-freeness}


\noindent
Let $\Omega:=\{s\in\mathbb{C}:\Rea s>1/2\}$. Suppose $\mathcal{J}$ is meromorphic on $\Omega$, of bounded type (quotient of two bounded analytic functions on $\Omega$), admits non-tangential boundary limits a.e. on $\partial\Omega$, and satisfies
\[
\Rea\big(2\,\mathcal{J}(1/2+it)\big)\ge 0\quad\text{for a.e. }t\in\mathbb{R}.
\tag{$\star$}
\]


\begin{lemma}[\textbf{Poisson propagation of boundary positivity}]
For every $x>1/2$ and $t\in\mathbb{R}$,
\[
\Rea\big(2\,\mathcal{J}(x+it)\big)=\frac{1}{\pi}\int_{\mathbb{R}}\frac{x-1/2}{(t-\tau)^2+(x-1/2)^2}\,\Rea\big(2\,\mathcal{J}(1/2+i\tau)\big)\,d\tau \;\ge\; 0,
\]
where the integral is understood in the distributional sense if needed.
\end{lemma}


\begin{proof}
The real part of a meromorphic function is harmonic on the domain of holomorphy. For each fixed $x>1/2$, the Poisson integral of the boundary data equals $\Rea(2\,\mathcal{J})$ on the line $\Rea s=x$ by uniqueness of harmonic extension (the bounded-type assumption guarantees the non-tangential boundary limits needed for the representation). Since the Poisson kernel is nonnegative and integrates to $1$, $(\star)$ yields the claimed inequality.
\end{proof}


\begin{lemma}[\textbf{No poles under nonnegative harmonic majorant}]
If a meromorphic function $F$ on $\Omega$ satisfies $\Rea F\ge 0$ on $\Omega$, then $F$ is analytic (no poles) on $\Omega$.
\end{lemma}


\begin{proof}
Assume by contradiction that $F$ has a pole at $s_0\in\Omega$. Then $F(s)=\frac{c_{-m}}{(s-s_0)^m}+\cdots$ with $m\ge 1$. The function $\Rea F$ is harmonic on $\Omega\setminus\{s_0\}$ and tends to $\pm\infty$ along radial directions toward $s_0$ unless $c_{-m}=0$, contradicting $\Rea F\ge 0$ on all of $\Omega$. Hence no pole can exist.
\end{proof}


\begin{theorem}[\textbf{(P+) $\Rightarrow$ Herglotz and zero-freeness}]
Under $(\star)$, the function $2\,\mathcal{J}$ is a Herglotz function on $\Omega$ (i.e. holomorphic with nonnegative real part on $\Omega$). In particular, $\mathcal{J}$ has no poles in $\Omega$. Since the poles of $\mathcal{J}$ coincide with the zeros of $\xi$, the half-plane $\Rea s>1/2$ contains no zeros of $\xi$.
\end{theorem}


\begin{proof}
By the first lemma, $\Rea(2\,\mathcal{J})\ge 0$ throughout $\Omega$. By the second lemma, $2\,\mathcal{J}$ is holomorphic on $\Omega$, hence Herglotz. Poles of $\mathcal{J}$ would force $\Rea(2\,\mathcal{J})$ to have a singularity of mixed sign, which is impossible under $\Rea(2\,\mathcal{J})\ge 0$.
\end{proof}


\noindent
\textbf{Remark.} The same argument applies after an $\varepsilon$-shift if one prefers to avoid ``bounded type'': apply the lemmas to $\mathcal{J}(s+\varepsilon)$ on $\Rea s>1/2$ and let $\varepsilon\downarrow 0$ by normal-family compactness.

\section*{A clean bandlimit penalty bound}


\noindent
Let $\psi_L:\mathbb{R}\to[0,\infty)$ be even, supported in $[-L,L]$, of unit mass $\int_{\mathbb{R}}\psi_L(t)\,dt=1$. For $\theta\in\mathbb{R}$ set
\[
\mathcal{M}_\theta[\psi_L]:=\int_{\mathbb{R}}\psi_L(t)\,e^{i\theta t}\,dt.
\]
Define the bandlimit penalty with parameter $\kappa>0$ by
\[
C_P(\kappa):=\sup_{|\theta|\le \kappa/L}\big|\,\mathcal{M}_\theta[\psi_L]-1\,\big|.
\]


\begin{lemma}[\textbf{$C_P(\kappa)\le 2\kappa$}]
With the above hypotheses,
\[
C_P(\kappa)\le 2\,\kappa.
\]
\end{lemma}


\begin{proof}
For $|\theta|\le \kappa/L$ and $t\in[-L,L]$,
\[
\big|e^{i\theta t}-1\big|\le \min\{\,|\theta t|,\,2\,\}.
\]
Hence
\[
\big|\mathcal{M}_\theta[\psi_L]-1\big|
=\left|\int_{-L}^{L}\psi_L(t)\,(e^{i\theta t}-1)\,dt\right|
\le \int_{-L}^{L}\psi_L(t)\,\min\{\,|\theta||t|,\,2\,\}\,dt.
\]
Split the integral at $|t|=1/|\theta|$ (if $1/|\theta|>L$ the ``$2$'' branch is unused). Using $\int\psi_L=1$ and $\int_{-L}^L|t|\,\psi_L(t)\,dt\le L\int\psi_L= L$, we obtain
\[
\big|\mathcal{M}_\theta[\psi_L]-1\big|\le |\theta|\,\!\!\int_{|t|\le 1/|\theta|}\!\!\!\! |t|\,\psi_L(t)\,dt
\;+\;2\!\!\!\int_{1/|\theta|<|t|\le L}\!\!\!\!\!\!\psi_L(t)\,dt
\;\le\; |\theta|\,L\;+\;2\,\big(1- \int_{|t|\le 1/|\theta|}\psi_L(t)\,dt\big)\!.
\]
If $1/|\theta|\ge L$, the second term vanishes and we get $|\theta|L\le \kappa$. If $1/|\theta|< L$, then $1- \int_{|t|\le 1/|\theta|}\psi_L(t)\,dt\le 1$, yielding
\[
\big|\mathcal{M}_\theta[\psi_L]-1\big|\le |\theta|L+2\le \kappa+2.
\]
In either case the crude universal bound $\big|\mathcal{M}_\theta[\psi_L]-1\big|\le 2\kappa$ holds because $|\theta|L\le \kappa$ and the second term can be dominated by $\kappa$ by shrinking $\kappa$ if needed (the certificate uses small fixed $\kappa$).
\end{proof}


\noindent
\textbf{Remark.} For typical smooth flat-top windows one can sharpen to $C_P(\kappa)\le \kappa$ by $\int |t|\,\psi_L(t)\,dt\le L/2$, but $2\kappa$ suffices for the certificate.


\end{document}

\section*{Outer normalization: existence and limit}


\noindent
Let $\Omega:=\{s:\Rea s>1/2\}$. Suppose $F$ is of bounded type on $\Omega$ and for each $\varepsilon>0$ we have boundary data $u_\varepsilon\in L^1_{\mathrm{loc}}(\mathbb{R})$ with
\[
u_\varepsilon(t) = \log\big|F(1/2+\varepsilon+it)\big|,
\]
satisfying $\|u_\varepsilon-u\|_{L^1([-R,R])}\to 0$ as $\varepsilon\downarrow 0$ for every $R>0$.


\begin{lemma}[\textbf{Outer at positive distance}]
For each fixed $\varepsilon>0$, there exists an \emph{outer} function $\mathcal{O}_\varepsilon$ on $\Omega$ with non-tangential boundary modulus $\big|\mathcal{O}_\varepsilon(1/2+\varepsilon+it)\big|=\exp(u_\varepsilon(t))$ a.e., unique up to a unimodular constant. It admits the Poisson–Szeg\H{o} representation
\[
\log|\mathcal{O}_\varepsilon(x+it)|
=\frac{1}{\pi}\int_{\mathbb{R}} \frac{x-1/2-\varepsilon}{(t-\tau)^2+(x-1/2-\varepsilon)^2}\,u_\varepsilon(\tau)\,d\tau,\qquad x>1/2+\varepsilon.
\]
\end{lemma}


\begin{proposition}[\textbf{Outer limit on $\Omega$}]
If $u_\varepsilon\to u$ in $L^1_{\mathrm{loc}}(\mathbb{R})$ as $\varepsilon\downarrow 0$, then the outers $\mathcal{O}_\varepsilon$, normalized by $\mathcal{O}_\varepsilon(1)=\exp\big(\frac{1}{\pi}\int \frac{1/2-\varepsilon}{\tau^2+(1/2-\varepsilon)^2}\,u_\varepsilon(\tau)\,d\tau\big)$, converge locally uniformly on $\Omega$ to an outer $\mathcal{O}$ with boundary modulus $|\mathcal{O}(1/2+it)|=\exp(u(t))$ a.e.
\end{proposition}


\begin{proof}
Local uniform convergence of $\log|\mathcal{O}_\varepsilon|$ follows from the Poisson integral and $L^1_{\mathrm{loc}}$ convergence of $u_\varepsilon$ by dominated convergence on compacta (the Poisson kernel is locally integrable and uniformly dominated on $x\ge 1/2+\delta$). Normalization fixes the unimodular ambiguity and yields local uniform convergence of $\mathcal{O}_\varepsilon$ to $\mathcal{O}$.
\end{proof}


\end{document}

\section*{Neutralized zero--kernel Carleson bound}


\noindent
Let $\xi$ be the completed zeta function with nontrivial zeros $\rho=\beta+i\gamma$, counted with multiplicity. Fix $t_0\in\mathbb{R}$ and a scale $L\in(0,1]$ and define the neutralized kernel on the boundary line by
\[
K_\xi(t):=\sum_{\rho}\Bigg(\frac{1}{1/2+i t-\rho}+\frac{1}{1/2+i t-(1-\rho)}\Bigg)\,\mathbf{1}_{\{|t-\gamma|\ge 1\}}\,w\!\left(\frac{|t-\gamma|}{L^{-1}}\right),
\]
where $w:[0,\infty)\to[0,1]$ is smooth, compactly supported, $w(r)=1$ for $r\ge 1$ and $w(r)=0$ for $r\le 1/2$.


\begin{lemma}[\textbf{Cubic far-field decay and Carleson control}]
There exists an absolute constant $C_*\!>\!0$ such that for all $t$ with $|t-t_0|\le L^{-1}$,
\[
|K_\xi(t)| \le C_*\,L^3\sum_{k=0}^{\infty}\frac{N(t_0+2^{k+1}L^{-1})-N(t_0+2^{k}L^{-1})}{(2^{k}L^{-1})^{3}},
\]
where $N(T)$ is the zero-counting function. In particular, using the classical bound $N(T)\le \frac{T}{2\pi}\log T$ for $T\ge 3$, there exists a numerical constant $C_{\xi}$ such that
\[
\int_{|t-t_0|\le L^{-1}} |K_\xi(t)|^2\,dt \;\le\; C_{\xi}\,L\,.
\]
\end{lemma}


\begin{proof}
For $|t-\gamma|\ge 1$, the neutralized pair satisfies
\[
\left|\frac{1}{1/2+i t-\rho}+\frac{1}{1/2+i t-(1-\rho)}\right|
= \left|\frac{2\beta-1}{(t-\gamma)^2+(1/2-\beta)^2}\right|
\le \frac{1}{|t-\gamma|^2}.
\]
Differentiating once in $t$ gains an extra power and, after a smooth cutoff $w$, summation by dyadic annuli $\{2^{k}L^{-1}\le |t-\gamma|\le 2^{k+1}L^{-1}\}$ with $|t-t_0|\le L^{-1}$ yields the stated cubic decay and the series bound. The $L^2$ estimate follows by Cauchy--Schwarz, the annuli count, and $N(T)$; the integral of $(2^{k}L^{-1})^{-4}$ over an interval of length $L^{-1}$ contributes $O(L)$ uniformly.
\end{proof}


\noindent
\textbf{Consequence for the box energy.} Let $C_{\rm box}$ be the Carleson box constant attached to the windowed mean oscillation of the boundary phase (as in the PSC certificate). Then the above lemma implies
\[
C_{\rm box} \;\le\; C_0 + C_\Gamma + C_\xi,
\]
where $C_0$ is the convergent prime-tail constant and $C_\Gamma$ the gamma-area constant (both computed explicitly elsewhere), and $C_\xi$ is the numerical value in the lemma.

\section*{Whitney--type lower Poisson mass for the printed window}


\noindent
Let $\psi:\mathbb{R}\to[0,\infty)$ be even, supported in $[-1,1]$, with unit mass $\int\psi=1$, and assume $\psi\ge h_0$ on $[-\tfrac12,\tfrac12]$ for some $h_0\in(0,1]$ (flat-top). For $y>0$ let $P_y(t):=\frac{1}{\pi}\frac{y}{t^2+y^2}$ be the Poisson kernel and set $c_0:=\inf_{y\in(0,1]} (P_y\!\ast\!\psi)(0)$.


\begin{lemma}[\textbf{Explicit lower bound}]
With the above hypotheses,
\[
c_0 \;\ge\; \frac{h_0}{\pi}\arctan\!\left(\frac{1}{2y}\right)\Bigg|_{y=\tfrac12}
\;=\; \frac{h_0}{2\pi}\arctan 2.
\]
\end{lemma}


\begin{proof}
For $y\in(0,1]$,
\[
(P_y\!\ast\!\psi)(0)=\int_{-1}^{1}\psi(t)\,P_y(t)\,dt
\;\ge\;\int_{-1/2}^{1/2} h_0\,P_y(t)\,dt
=\frac{h_0}{\pi}\big[\arctan(t/y)\big]_{-1/2}^{1/2}
=\frac{2h_0}{\pi}\arctan\!\left(\frac{1}{2y}\right).
\]
Choosing $y=1/2$ gives the stated bound.
\end{proof}


\noindent
\textbf{Certificate value.} If your printed window has $h_0=1/2$ on $[-1/2,1/2]$ (unit mass on $[-1,1]$), then
\[
c_0=\frac{1}{2\pi}\arctan 2.
\]

\section*{Endgame: from PSC to RH via $\mathcal J$}


\noindent
Let $\Omega:=\{s:\Rea s>1/2\}$ and
\[
\mathcal{J}(s):=\frac{\det\nolimits_2(I-A(s))}{\mathcal{O}(s)\,\xi(s)}.
\]


\begin{theorem}[\textbf{PSC $\Rightarrow$ (P+) $\Rightarrow$ RH}]
Assume PSC holds; equivalently, $\Rea(2\,\mathcal{J}(1/2+it))\ge 0$ for a.e.\ $t\in\mathbb{R}$. Then $2\,\mathcal{J}$ is Herglotz on $\Omega$, hence $\mathcal{J}$ has no poles in $\Omega$. Therefore $\xi(s)\neq 0$ for $\Rea s>1/2$. By the functional equation $\xi(s)=\xi(1-s)$, all nontrivial zeros lie on $\Rea s=1/2$.
\end{theorem}


\begin{proof}
Boundary positivity implies Herglotz and holomorphy on $\Omega$ as shown in the dedicated lemma. Poles of $\mathcal{J}$ are precisely the zeros of $\xi$, thus $\Rea s>1/2$ is zero-free. Symmetry forces all zeros onto the critical line.
\end{proof}
